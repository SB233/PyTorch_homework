class LSTMNetwork(nn.Module):
    def __init__(self, input_size, hidden_size, output_size, n_layers=1):
        super(LSTMNetwork, self).__init__()
        self.n_layers = n_layers
        self.hidden_size = hidden_size

        # LSTM
        # embeddinglist
        self.embedding = nn.Embedding(input_size, hidden_size)
        # LSTMhidden_sizeLSTMn_layers
        self.lstm = nn.LSTM(hidden_size, hidden_size, n_layers)
        # softmax
        self.fc = nn.Linear(hidden_size, output_size)
        self.logsoftmax = nn.LogSoftmax()

    def forward(self, input, hidden=None):
        #input
        embedded = self.embedding(input)
        
        # 
        # PyTorchLSTM
        # batch_sizeembedded
        # batchbatch_size1
        # input_list_size, batch_size, hidden_size
        embedded = embedded.view(input.data.size()[0], 1, self.hidden_size)
    
        # PyTorchLSTM
        # outputhidden
        # hiddentuplecell
        
        output, hidden = self.lstm(embedded, hidden)
        
        #
        output = output[-1,...]

        #
        out = self.fc(output)
        # softmax
        out = self.logsoftmax(out)
        return out

    def initHidden(self):
        # 
        # 0.
        # hiddencelllayers,batch_size,hidden_size
        hidden = Variable(torch.zeros(self.n_layers, 1, self.hidden_size))
        # cell0
        cell = Variable(torch.zeros(self.n_layers, 1, self.hidden_size))
        return (hidden, cell)

records = []

# 
def time_since(since):
    now = time.time()
    s = now - since
    m = math.floor(s / 60)
    s -= m * 60
    return '%dm %ds' % (m, s)

start = time.time()

# 5epoch
for epoch in range(5):
    losses = []
    # EPOCH 
    for i in range(all_line_num):
        category, line, y, x = random_training_pair()
        x = Variable(torch.LongTensor(x))
        y = Variable(torch.LongTensor(np.array([y])))
        optimizer.zero_grad()
        
        # Step1:LSTM
        hidden = lstm.initHidden()
        
        # Step2:LSTMPyTorchLSTM
        # 
        output = lstm(x, hidden)
        
        # Step3:
        loss = cost(output, y)
        losses.append(loss.data.numpy()[0])
        
        #
        loss.backward()
        optimizer.step()
        
        #3000
        if i % 3000 == 0:
            # 
            guess, guess_i = category_from_output(output)
            correct = '' if guess == category else ' (%s)' % category
            # 
            training_process = (all_line_num * epoch + i) / (all_line_num * 5) * 100
            training_process = '%.2f' % training_process
            print('{}{:.2f}{}%{}{}{}{}'\
                .format(epoch, np.mean(losses), float(training_process), time_since(start), line, guess, correct))
            records.append([np.mean(losses)])
