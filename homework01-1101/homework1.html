# 10020030

input_size = 100
output_size = 200
hidden_size = 30

# 
model = nn.Sequential(nn.Linear(input_size, hidden_size),
                    nn.Tanh(),
                    nn.Linear(hidden_size, output_size)
                    )

# 
criterion = torch.nn.MSELoss()

# 
optimizer = torch.optim.Adam(model.parameters(), lr = 0.0001)

# 
num_epoch = 100


#500
results = []
for epoch in range(num_epoch):
    train_loss = []
    for data in train_data:
        # 
        x = Variable(torch.FloatTensor(data[0])).unsqueeze(0)
        y = Variable(torch.FloatTensor(data[1])).unsqueeze(0)
        # 
        output = model(x)
        
        # 
        optimizer.zero_grad()
        loss = criterion(output, y)
        train_loss.append(loss.data.numpy()[0])
        loss.backward()
        optimizer.step()
    # 
    valid_loss = []
    for data in valid_data:
        x = Variable(torch.FloatTensor(data[0])).unsqueeze(0)
        y = Variable(torch.FloatTensor(data[1])).unsqueeze(0)
        output = model(x)
        loss = criterion(output, y)
        valid_loss.append(loss.data.numpy()[0])
    results.append([np.mean(train_loss), np.mean(valid_loss)])
    print('{}Loss: {:.2f}, Loss: {:.2f}'.format(epoch, np.mean(train_loss), np.mean(valid_loss)))
